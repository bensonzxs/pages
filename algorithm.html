                    }

                }

二叉树的中序遍历

            
                vector inorderTraversal(TreeNode *root) {
                    vector res;
                    if(root == NULL)
                        return res;
                    stack s;
                    while(root != NULL || !s.empty()){
                        while(root != NULL){
                            s.push(root);
                            root = root->left;
                        }
                            root = s.top();
                            res.emplace_back(root->val);
                            s.pop();
                            root = root->right;
                    }
                    return res; 
                }

                vector postorderTraversal(TreeNode *root) {
                    vector res;
                    if(root == nullptr)
                        return res;
                    stack st;
                    TreeNode *work = prev;
                    while(!st.empty() || root != nullptr){
                        while(root != nullptr){
                            st.push(root);
                            root = root->left;
                        }
                        root = st.top();
                        st.pop();
                        if(root->right == nullptr || root->right == prev){
                            res.push_back(root->data);
                            prev = root;
                            root = nullptr;
                        }
                        else{
                            st.push(root); // 根结点重新入栈 
                            root = root->right; //遍历右子树
                        } 
                    }
                    return res;
                }
